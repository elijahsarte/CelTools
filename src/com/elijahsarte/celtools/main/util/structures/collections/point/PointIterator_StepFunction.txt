package com.elijahsarte.celtools.main.util.structures.collections.point;

import com.elijahsarte.celtools.main.util.ProgrammingEx;
import com.elijahsarte.celtools.main.util.datastructures.EnhancedTreeMap;
import com.elijahsarte.celtools.main.util.structures.collections.IntList;

import java.awt.*;
import java.util.Iterator;
import java.util.NoSuchElementException;
import java.util.function.Consumer;

public class PointIterator implements Iterator<Point> {

    private final PointCollection ptColl;
    private int startIndex, index = 0;
    private EnhancedTreeMap.Entry<Integer, IntList> currEntry;

    public PointIterator(PointCollection ptColl) {
        this.ptColl = ptColl;
    }

    @Override
    public boolean hasNext() {
        return (index + 1) < ptColl.size();
    }

    private Point toPt(int yIdx) {
        return new Point(currEntry.getKey(), currEntry.getValue().get(yIdx));
    }
    private void fixPrev() {
        this.currEntry = EnhancedTreeMap.predecessor(currEntry);
        this.startIndex = startIndex - currEntry.getValue().size();
    }
    @Override
    public Point next() {
        if (!hasNext()) throw new NoSuchElementException();
//        if (ProgrammingEx.anyEquals(ptColl.getEntry(index).getKey(), 320, 321, 322, 323, 324, 325, 326, 327, 328, 329)) {
//        if (ProgrammingEx.anyEquals(ptColl.getEntry(index).getKey(), 339, 340)) {
//            int j = 9;
//        }
        if (this.currEntry == null) {
            ptColl.onRaw();
            this.currEntry = ptColl.getEntry(index);
            this.startIndex = index;
        }
        if ((index - startIndex) < 0) fixPrev();
        else if ((index - startIndex) >= currEntry.getValue().size()) {
            this.currEntry = EnhancedTreeMap.successor(this.currEntry);
            this.startIndex = index;
        }
        if (this.currEntry != ptColl.getEntry(index)) {
            int j = 9;
        }
        return toPt(index++ - startIndex);
    }

    @Override
    public void remove() {
        if (index - startIndex == 0) fixPrev();
        currEntry.getValue().remove(--index - startIndex);
        if (currEntry.getValue().isEmpty()) {
            ptColl.removeEntry(currEntry);
            this.currEntry = null;
        }
    }
    public void removeX() {
        index = startIndex;
        this.currEntry = null;
    }


    @Override
    public void forEachRemaining(Consumer<? super Point> action) {
        Iterator.super.forEachRemaining(action);
    }
}
